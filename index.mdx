<Danger>
  Client is personalized for your account. Do not share clients with others
</Danger>

<Steps>
  <Step title="Log in">
    Auth in https://auth.assembly.lol/login
  </Step>
  <Step title="Download the client">
    On the page after logging in, click Download Client and save the archive
  </Step>
  <Step title="Prepare Roblox">
    Launch Roblox and wait until the game window appears (the client must detect the active window)
  </Step>
  <Step title="Install">
    Unzip the archive to a convenient folder. Open client.exe as administrator
  </Step>
  <Step title="Done">
    Wait for the client to connect. If you see any errors - go to [Help & Support](/help) section
  </Step>
</Steps>
